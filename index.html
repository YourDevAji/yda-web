          try {
            final response = await http.get(Uri.parse('https://user/data'));

            if (response.statusCode == 200) {
              print(response.body);
            } else {
              print('Failed to load profile data.');
            }
          } catch (error) {
            print('Error fetching profile: $error');
          }
        }
